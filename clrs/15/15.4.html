def lcs(X, Y):
    X, Y = (X, Y) if len(X) < len(Y) else (Y, X)
    n = len(X)
    m = len(Y)
    C = [0] * n
    j = 0
    a = 0
    for i in range(m):
        b = C[i]
        if X[i] == Y[j]:
            C[i] = a + 1
        else:
            C[i] = max(C[i-1] if i > 0 else 0, C[i])
        a = b
    return C[n]
